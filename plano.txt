CREATE TABLE planos_acao (
  -- Coluna de identificação padrão e data de criação
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),

  -- Informações do Cliente (Seção Header e 1)
  -- Armazena o nome da clínica e o nome da pessoa de contato.
  nome_clinica TEXT NOT NULL,
  nome_contato TEXT,

  -- Definição da Estratégia (Seção 2)
  -- Usamos um array de texto para os serviços, permitindo flexibilidade (3, 4, 5 ou mais serviços).
  servicos_foco TEXT[],
  -- Armazena a escolha de posicionamento: 'Volume' ou 'Exclusividade'.
  posicionamento_mercado TEXT,
  -- Campo de texto livre para descrever o diferencial da clínica.
  diferencial TEXT,

  -- Plano de Investimento (Seção 3)
  -- Usamos o tipo NUMERIC para valores financeiros para garantir a precisão.
  investimento_gestao NUMERIC,
  investimento_anuncios NUMERIC,

  -- Checklist de Próximos Passos (Seção 4)
  -- Usamos JSONB para o checklist. É a forma mais flexível de armazenar
  -- o estado de cada item sem precisar de uma coluna para cada um.
  checklist_status JSONB,
  
  -- Informações do Gestor de Tráfego (Rodapé)
  -- Armazena os dados de quem está prestando o serviço.
  gestor_nome TEXT,
  gestor_contato TEXT
);

-- Adicionando comentários nas colunas para melhor entendimento dentro do Supabase.
COMMENT ON COLUMN planos_acao.nome_clinica IS 'Nome da clínica cliente, ex: Clínica Florê';
